def closeEvent(self, event):
    if self.is_dirty:
        reply = QMessageBox.question(self, '退出',
                                   '文件未保存，是否退出？',
                                   QMessageBox.StandardButton.Yes |
                                   QMessageBox.StandardButton.No |
                                   QMessageBox.StandardButton.Cancel)
        if reply == QMessageBox.StandardButton.Yes:
            event.accept()
        else:
            event.ignore()
    else:
        event.accept()
